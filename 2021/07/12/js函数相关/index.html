    限制input输入两位小数的正则

    <input :value="form.price" @input="onlyFloat($event, 'price')"/>

methods: {
    onlyFloat(e, type) {
        let val = e.detail.value
        val = val.replace(/^(\-)*(\d+)\.(\d\d).*$/, '$1$2.$3')
        this.form[type] = val
        return val
    }
}
